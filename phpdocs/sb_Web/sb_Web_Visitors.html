Class sb_Web_Visitors

Description
Description | Vars (details) | Methods (details)

Keeps track of online users with an in memory mysql table

  1.  //Requires the following SQL change myDatabase database to a database you want to use, if you already have one, skip the create step and just use it.  replace @myDatabase, @myUser, @myPass and @myHost with the appropriate data for your project
  2.  //Make sure to change the length of uname VARCHAR to the longest uname you will store
  3.  CREATE DATABASE @myDatabase;
  4.    USE @myDatabase;
  5.    CREATE TABLE online_visitors(
  6.       id INT UNSIGNED NOT NULL AUTO_INCREMENT,
  7.       mobl CHAR(1) DEFAULT 0,
  8.       ip INT,
  9.       tstamp INT(10) UNSIGNED,
  10.       uname VARCHAR(15),
  11.       dname VARCHAR(50),
  12.       agent VARCHAR(50),
  13.       agent_str VARCHAR(500),
  14.       PRIMARY KEY (id)
  15.   ) ENGINE = MEMORY;
  16.  
  17.   GRANT ALL ON @myDatabase.* TO '@myUser'@'@myHost' IDENTIFIED BY '@myPass';
  18.  
  19.  $visitors = sb_Web_Visitors::get_visitor_data();
  20.  echo '<b title="'.implode(",", sb_Web_Visitors::list_users()).'">(?) </b>users: '.$visitors->users.'/ guests: '.$visitors->guests.'/ bots:'.$visitors->bots;
